\exercise[1.2 (Disjunctive Random Variables)]
\begin{align*}
	P(A\lor B \lor C) &= P(A\lor B) + P(C) - P((A\lor B)\land C)  \\
	&= P(A) + P(B) - P(A\land B) + P(C) - P(A\land C \lor B \land C) \\
	&= P(A) + P(B) + P(C) - P(A\land B) - P(A\land C) - P(B \land C) + P(A\land B \land C)
\end{align*}
